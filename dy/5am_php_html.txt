function Spider_RAND($length)
{
	$possible = "ABCDEFGHIJKLM1234567890NOPQR";
	$str = "";
	while(strlen($str) < $length)
	$str .= substr($possible,(rand() % strlen($possible)),1);
	return $str;
}
